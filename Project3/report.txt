    //Terms
    //Valid Direction: 'u', 'U', 'd', 'D', 'l', 'L', 'r', 'R'
    isSyntacticallyCorrect function {
        if string is empty,
            return true
        repeatedly:
            if character is not a freeze, valid direction, or beat,
                return false;
            else if character is beat,
                increment one index
            else if direction followed by beat,
                increment two indices
            else if single digit freeze followed by direciton followed by beat,
                increment three indices
            else if double digit freeze followed by direction followed by beat,
                increment four indices
            else
                return false
        return true
    }

    translateDance function {
        copy orignal instructions
        clear instructions
        if not syntatically correct,
            instructions become original instructions
            return 1
        repeatedly:
            //sections of dances are seperated by a valid direction, freeze, or beat
            find next section of the dance 
            if freeze,
                if freeze length is less than 2,
                    bad beat is the next beat
                    return 2
                repeatedly for freeze amount of times unless is end of string:
                    if beat is not slash,
                        instructions become original instructions
                        bad beat is the next beat
                        return 3
                    add uppercase direction to instructions
                    increment one index and beat
                if number of beats is not equal to freeze amount,
                    instructions become original instructions
                    bad beat is next beat
                    return 4
                increment three + number of freeze amount indices
                increment number of freeze amount of beats
            else if direction,
                translate to lowercase version of direction
                increment two indices and beats
            else beat,
                translate to a dot
                increment one index and beat
        return 0;
    }
